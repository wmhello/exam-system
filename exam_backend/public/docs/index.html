    "http://api.exam.test/api/admin/sms/send"
);

let headers = {
    "Content-Type": "application/json",
    "Accept": "application/json",
    "Authorization": "Bearer {{token}}",
};

let body = {
    "phone": "ullam"
}

fetch(url, {
    method: "POST",
    headers: headers,
    body: body
